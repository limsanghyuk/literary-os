### check_version_consistency.py 6 파일 검사 통합

```python
# tools/check_version_consistency.py (확장)
"""
SSoT: pyproject.toml [project].version
검사 대상: README 뱃지 / CHANGELOG / MANIFEST / ci.yml Gates / git tag
"""
from pathlib import Path
import re, sys, tomllib

ROOT = Path(__file__).parent.parent

def get_pyproject_version() -> str:
    data = tomllib.loads((ROOT / 'pyproject.toml').read_text())
    return data['project']['version']

def get_gate_count() -> int:
    from literary_system.gates.release_gate_registry import RELEASE_GATE_REGISTRY
    return len(RELEASE_GATE_REGISTRY.all())

CHECKS = [
    {'name': 'pyproject.toml',
     'pattern': r'version\s*=\s*["\'](\d+\.\d+\.\d+)["\']',
     'expected_fn': get_pyproject_version},
    {'name': 'README.md (version badge)',
     'file': 'README.md',
     'pattern': r'version-(\d+\.\d+\.\d+)-blue',
     'expected_fn': get_pyproject_version},
    {'name': 'README.md (gate badge)',
     'file': 'README.md',
     'pattern': r'release%20gates-(\d+)%2F(\d+)',
     'expected_fn': lambda: (get_gate_count(), get_gate_count())},
    {'name': 'CHANGELOG.md (latest)',
     'file': 'CHANGELOG.md',
     'pattern': r'## \[(\d+\.\d+\.\d+)\]',
     'expected_fn': get_pyproject_version,
     'match_mode': 'first'},
    {'name': 'MANIFEST.md',
     'file': 'MANIFEST.md',
     'pattern': r'Version:\s*(\d+\.\d+\.\d+)',
     'expected_fn': get_pyproject_version},
    {'name': '.github/workflows/ci.yml (gate count)',
     'file': '.github/workflows/ci.yml',
     'pattern': r'Release Gate \((\d+) Gates\)',
     'expected_fn': get_gate_count},
]
```

### E2EProseGate 6 checkpoints

```python
# literary_system/gates/e2e_prose_gate.py (신규)
from dataclasses import dataclass
from typing import Literal

@dataclass
class CheckpointResult:
    cp_id: str
    name: str
    status: Literal['PASS', 'FAIL', 'SKIP']
    elapsed_ms: int
    details: dict

class E2EProseGate:
    MODE_MOCK = 'mock'
    MODE_REAL = 'real'

    CHECKPOINTS = [
        ('CP-1', 'NIE NIL 6단계 통과'),
        ('CP-2', 'ASD AutoRepair (debt + Gate26/27)'),
        ('CP-3', 'GIG NarrativeGraph + BlastRadius ≤ 0.55'),
        ('CP-4', 'LOSDB QueryInterface HEALTHY + ≤ 1초'),
        ('CP-5', 'Constitution R(scene) ≥ 0.65'),
        ('CP-6', 'Minimal-CLI generate 100~500자'),
    ]

    def run(self, mode: str = MODE_MOCK) -> list[CheckpointResult]:
        results = []
        for cp_id, name in self.CHECKPOINTS:
            method = getattr(self, f'_run_{cp_id.lower().replace("-", "_")}')
            results.append(method(mode))
        return results
```

### Gate Registry tier 메타

```python
# literary_system/gates/release_gate_registry.py — 확장
@dataclass
class GateMetadata:
    id: str
    name: str
    func: Callable
    tier: Literal['L0', 'L1', 'L2', 'L3'] = 'L3'   # ★ V587 신규
    typical_elapsed_sec: float | None = None       # ★ V587 신규
    historical_block_rate: float = 0.0              # ★ V587 신규

def register_gate(id: str, name: str, tier: str = 'L3'):
    def decorator(fn):
        RELEASE_GATE_REGISTRY.register(GateMetadata(
            id=id, name=name, func=fn, tier=tier,
        ))
        return fn
    return decorator

# 후방 호환: tier 기본값 = 'L3' → 기존 게이트 미수정 시 L3 (안전)
```
